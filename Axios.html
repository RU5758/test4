Chart.js

Chart.js 圖表

程式碼範例

                <div id="chart"></div>
                <script>
                  const data = [10, 15, 20, 25, 30];
                
                  const width = 500;
                  const height = 200;
                
                  const svg = d3.select("#chart")
                                .append("svg")
                                .attr("width", width)
                                .attr("height", height);
                
                  svg.selectAll("rect")
                     .data(data)
                     .enter()
                     .append("rect")
                     .attr("x", (d, i) => i * 50)
                     .attr("y", (d) => height - d * 5)
                     .attr("width", 40)
                     .attr("height", (d) => d * 5)
                     .attr("fill", "blue");
                </script>
